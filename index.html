// === Thai Address Auto-fill Algorithm ===

FUNCTION onZipCodeInput(zipCode):
    // Step 1: Search matching zip codes
    results = searchZipCodes(zipCode)
    
    IF results.length > 0:
        showSuggestions(results)
    ELSE:
        hideSuggestions()

FUNCTION onZipCodeSelected(zipCode):
    // Step 2: Filter geography data by zip
    filteredData = filterByZipCode(zipCode)
    
    // Step 3: Extract unique provinces
    provinces = getUniqueProvinces(filteredData)
    populateDropdown("province", provinces)
    
    // Step 4: Auto-select if only one option
    IF provinces.length == 1:
        selectProvince(provinces[0])
        onProvinceChange()

FUNCTION onProvinceChange(provinceCode):
    // Cascade: Filter districts
    districts = filterByProvince(provinceCode)
    populateDropdown("district", districts)
    
    IF districts.length == 1:
        onDistrictChange(districts[0])

FUNCTION onDistrictChange(districtCode):
    // Cascade: Filter subdistricts
    subdistricts = filterByDistrict(districtCode)
    populateDropdown("subdistrict", subdistricts)

// === Edge Case: Manual Edit ===
FUNCTION onManualEdit(field, value):
    // Switch to MANUAL_MODE
    mode = MANUAL_MODE
    dataSource = fullGeographyData
    
    IF field == "province":
        updateDistrictOptions()
        clearSubdistrict()
    ELSE IF field == "district":
        updateSubdistrictOptions()
    ELSE IF field == "subdistrict":
        // Sync zip code with selected area
        updateZipCode(subdistrict.postalCode)
